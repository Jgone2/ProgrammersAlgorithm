class Solution {
    public int[] solution(int[] arr) {
        int idx = 1;
        int arrIdx = arr.length;
        while(idx < arr.length) {
            idx <<= 1;
        }
        int[] answer = new int[idx];
        for(int i = 0; i < idx; i++) {
            if(i < arrIdx) answer[i] = arr[i];
            else answer[i] = 0;
        }
        return answer;
    }
}
