class Solution {
    public int[] solution(int[] num_list, int n) {
        int[] answer = new int[num_list.length];

class Solution {
    public int[] solution(int[] num_list, int n) {
        int idx = 0;
        int[] answer = new int[num_list.length];
        for (int i = n;i < num_list.length;i++)
            answer[idx++] = num_list[i];

            answer[idx++] = num_list[i];
        return answer;
    }
