class Solution {
    public int solution(int[] num_list) {
        int answer = 1;
        if (num_list.length > 10) {
            answer--;
            for (int e : num_list) {
                answer += e;
            }
        } else {
            for (int e : num_list) {
                answer *= e;
            }
        }
        return answer;
    }
}
