class Solution {
    public int solution(int[] numbers) {
        int answer = 0;
        boolean[] check = new boolean[10];

        for (int number : numbers) {
            if (number >= 0 && number <= 9) {

            }
        }

            if (!check[i]) {
                answer += i;
            }
        }

        return answer;
    }
}
