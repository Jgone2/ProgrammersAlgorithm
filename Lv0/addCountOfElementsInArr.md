class Solution {
    public int[] solution(int[] arr) {
        int size = 0;
        for (int e : arr) size += e;
        
        int[] answer = new int[size];
        int idx = 0;
        for (int e : arr) {
            for (int i = 0; i < e; i++) {
                answer[idx++] = e;
            }
        }
        return answer;
    }
}

import java.util.stream.Stream;

class Solution {
    public int[] solution(int[] arr) {
        String answer = "";

        for(int i: arr) {
            answer += (String.valueOf(i)+",").repeat(i);
        }
        return Stream.of(answer.split(",")).mapToInt(Integer::parseInt).toArray(); 
    }
}
