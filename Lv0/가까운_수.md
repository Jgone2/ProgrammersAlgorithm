class Solution {
    public int solution(int[] array, int n) {
        int answer = array[0];
        int min = Math.abs(array[0] - n);
        for(int i = 1; i < array.length; i++) {
            if(Math.abs(array[i] - n) < min) {
                answer = array[i];
                min = Math.abs(array[i] - n);
            }
            else if(Math.abs(array[i] - n) == min && array[i] < answer)             {
                answer = array[i];
            }
        }
        return answer;
    }
}
